### 3. 详细部署步骤

#### 3.1 服务器 1 - 初始化

```bash
# 1. 更新系统
sudo apt update && sudo apt upgrade -y

# 2. 安装 Node.js
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo -E bash -
sudo apt install -y nodejs

# 3. 安装 Docker
curl -fsSL https://get.docker.com -o get-docker.sh
sudo sh get-docker.sh
sudo usermod -aG docker $USER

# 4. 安装 Redis
sudo apt install -y redis-server

# 5. 配置 Redis Master
sudo nano /etc/redis/redis.conf

# 修改以下配置：
# bind 0.0.0.0                    # 允许远程连接
# port 6379
# requirepass your_password        # 设置密码
# appendonly yes                   # 启用 AOF 持久化
# appendfsync everysec

# 6. 启动 Redis
sudo systemctl restart redis-server
sudo systemctl enable redis-server

# 7. 安装 PM2
sudo npm install -g pm2

# 8. 克隆项目
git clone <your-repo> /opt/latex-compiler
cd /opt/latex-compiler

# 9. 安装依赖
npm install

# 10. 配置环境变量
cat > .env << EOF
NODE_ENV=production
REDIS_HOST=localhost
REDIS_PORT=6379
REDIS_PASSWORD=your_password
REDIS_DB=0
COMPILE_TIMEOUT=30
WORKER_COUNT=2
API_PORT=3001
EOF

# 11. 启动服务
pm2 start ecosystem.config.js
pm2 save
pm2 startup
```

#### 3.2 服务器 2 - 初始化

```bash
# 1-3 步同服务器 1

# 4. 安装 Redis
sudo apt install -y redis-server

# 5. 配置 Redis Slave
sudo nano /etc/redis/redis.conf

# 修改以下配置：
# bind 0.0.0.0
# port 6379
# requirepass your_password
# slaveof server1_ip 6379          # 指向 Master
# masterauth your_password          # Master 密码
# appendonly yes
# appendfsync everysec

# 6. 启动 Redis
sudo systemctl restart redis-server
sudo systemctl enable redis-server

# 7-11 步同服务器 1，但环境变量中：
# REDIS_HOST=server1_ip            # 指向 Master
```

### 5. 核心代码实现

#### 5.1 Redis 连接配置

```typescript
// src/config/redis.ts
import Redis from 'ioredis';

// 创建 Redis 客户端
export const redisClient = new Redis({
  host: process.env.REDIS_HOST || 'localhost',
  port: parseInt(process.env.REDIS_PORT || '6379'),
  password: process.env.REDIS_PASSWORD,
  db: parseInt(process.env.REDIS_DB || '0'),
  retryStrategy: (times) => {
    const delay = Math.min(times * 50, 2000);
    return delay;
  },
  enableReadyCheck: false,
  enableOfflineQueue: true
});

// 监听连接事件
redisClient.on('connect', () => {
  console.log('Redis connected');
});

redisClient.on('error', (err) => {
  console.error('Redis error:', err);
});

export default redisClient;
```

#### 5.2 编译队列配置

```typescript
// src/queue/compileQueue.ts
import Queue from 'bull';
import redisClient from '../config/redis';

// 创建编译队列
export const compileQueue = new Queue('compile', {
  redis: {
    host: process.env.REDIS_HOST,
    port: parseInt(process.env.REDIS_PORT || '6379'),
    password: process.env.REDIS_PASSWORD
  },
  defaultJobOptions: {
    attempts: 3,  // 失败重试 3 次
    backoff: {
      type: 'exponential',
      delay: 2000
    },
    removeOnComplete: true,
    removeOnFail: false
  }
});

// 监听队列事件
compileQueue.on('completed', (job) => {
  console.log(`Job ${job.id} completed`);
});

compileQueue.on('failed', (job, err) => {
  console.error(`Job ${job.id} failed:`, err.message);
});

export default compileQueue;
```

#### 5.3 Worker 进程

```typescript
// src/workers/compileWorker.ts
import compileQueue from '../queue/compileQueue';
import { compileLatex } from '../services/compileService';

// 处理编译任务
compileQueue.process(
  parseInt(process.env.WORKER_COUNT || '2'),
  async (job) => {
    console.log(`Processing job ${job.id} on ${process.env.HOSTNAME}`);
    
    try {
      const result = await compileLatex(job.data);
      
      // 保存结果到 Redis
      await redisClient.setex(
        `compile:${job.id}`,
        3600,  // 1 小时过期
        JSON.stringify(result)
      );
      
      return result;
    } catch (error) {
      console.error(`Job ${job.id} error:`, error);
      throw error;
    }
  }
);

console.log(`Worker started on ${process.env.HOSTNAME}`);
```

#### 5.4 API Gateway

```typescript
// src/api/server.ts
import express from 'express';
import compileQueue from '../queue/compileQueue';

const app = express();
app.use(express.json());

// 提交编译任务
app.post('/api/compile', async (req, res) => {
  try {
    const { source, engine, format } = req.body;
    
    // 添加任务到队列
    const job = await compileQueue.add(
      {
        source,
        engine: engine || 'pdflatex',
        format: format || 'pdf'
      },
      {
        priority: 10  // 优先级
      }
    );
    
    res.json({
      jobId: job.id,
      status: 'queued'
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 获取编译结果
app.get('/api/compile/:jobId', async (req, res) => {
  try {
    const { jobId } = req.params;
    
    // 从 Redis 获取结果
    const result = await redisClient.get(`compile:${jobId}`);
    
    if (result) {
      res.json(JSON.parse(result));
    } else {
      // 从队列获取任务状态
      const job = await compileQueue.getJob(jobId);
      
      if (!job) {
        return res.status(404).json({ error: 'Job not found' });
      }
      
      const state = await job.getState();
      const progress = job.progress();
      
      res.json({
        jobId,
        status: state,
        progress
      });
    }
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// 健康检查
app.get('/health', (req, res) => {
  res.json({ status: 'ok', hostname: process.env.HOSTNAME });
});

app.listen(3001, () => {
  console.log('API Gateway listening on port 3001');
});
```

### 6. 负载均衡配置

#### 6.1 使用 Nginx 做反向代理

```nginx
# /etc/nginx/sites-available/latex-compiler
upstream compile_backend {
    least_conn;  # 最少连接算法
    server server1_ip:3001 weight=1;
    server server2_ip:3001 weight=1;
    keepalive 32;
}

server {
    listen 80;
    server_name api.latex-editor.com;

    location /api/ {
        proxy_pass http://compile_backend;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        
        # 超时配置
        proxy_connect_timeout 5s;
        proxy_send_timeout 30s;
        proxy_read_timeout 30s;
    }
}
```

### 7. 监控和告警

#### 7.1 监控脚本

```typescript
// src/monitoring/monitor.ts
import redisClient from '../config/redis';
import compileQueue from '../queue/compileQueue';

async function monitorSystem() {
  setInterval(async () => {
    try {
      // 获取队列统计
      const counts = await compileQueue.getJobCounts();
      const waiting = counts.waiting;
      const active = counts.active;
      const completed = counts.completed;
      const failed = counts.failed;
      
      // 获取 Redis 信息
      const info = await redisClient.info('memory');
      
      console.log(`
        Queue Status:
        - Waiting: ${waiting}
        - Active: ${active}
        - Completed: ${completed}
        - Failed: ${failed}
        
        Redis Memory: ${info}
      `);
      
      // 告警条件
      if (waiting > 100) {
        console.warn('⚠️ Queue backlog too high!');
        // 发送告警
      }
      
      if (failed > 10) {
        console.error('❌ Too many failed jobs!');
        // 发送告警
      }
    } catch (error) {
      console.error('Monitor error:', error);
    }
  }, 60000);  // 每分钟检查一次
}

monitorSystem();
```

## 总结

这个分布式方案的优势：

✅ **高可用**: Redis Master-Slave 自动故障转移
✅ **高性能**: 4 个 Worker 并发编译
✅ **易扩展**: 可轻松添加更多服务器
✅ **成本低**: 充分利用两台服务器
✅ **易维护**: PM2 自动管理进程
